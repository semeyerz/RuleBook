
The maximum time for this test is 5 minutes, 5 more for the optional guiding-phase.

\begin{scorelist}

	% Up to 30 points
	\scoreheading{Waypoint 1: Path planning}
	\scoreitem{30}{Reaching waypoint 1}

	% Up to 40 points
	\scoreheading{Waypoint 2: Obstacle Interaction 1}
	\scoreitem{10}{Correctly state the nature of the obstacle blocking its path}
	\scoreitem{30}{Reaching waypoint 2 (to grasp/place distance)}

	% Up to 40 points
	\scoreheading{Waypoint 3: Obstacle Interaction 2}
	\scoreitem{10}{Correctly state the nature of the obstacle blocking its path}
	\scoreitem{30}{Reaching waypoint 3}
	
	% Up to 50 points
	\scoreheading{Obstacle avoidance}
	\scoreitem{10}{Avoiding 3D object (Difficult-to-see object)}
	\scoreitem{10}{Asking a person to step aside}
	\scoreitem{30}{Moving away movable object}
	%penalties for collisions
	\scoreitem{-5}{colliding with 5x5x5 object}
	\scoreitem{-10}{colliding with 10x10x10 object}
	\scoreitem{-15}{colliding with 15x15x15 object}

	% Up to 40 points
	\scoreheading{Leaving the arena}
	\scoreitem{5}{Reaching the door via which the robot went out the apartment again. The handle must be within manipulator reach}
	\scoreitem{15}{Pushing the door handle/knob/lever with manipulator}
	\scoreitem{15}{Opening the door far enough to drive in}
	\scoreitem{5}{Leaving the arena}

	\setTotalScore{200}
\end{scorelist}
